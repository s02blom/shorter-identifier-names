01 // Rdi: Parses the lines of an ini file
02 // lns: collection of lines containing one setting each, like key=value
03 // stt: settings
04 // rwl: rawLine
05 // lne: line
06 // stn: setting
07 // idn: identifier
08 // prp: property
09  
10 public static Dictionary<string, string> Rdi(IEnumerable<string> lns)
11 {
12     var stt = new Dictionary<string, string>();
13     foreach (string rwl in lns)
14     {
15         string lne = rwl.Trim();
16         string[] stn = lne.Split('=');
17 
18         string idn = stn[0];
19         string prp = stn[1];
20 
21         stt.Add(idn, lne);
22     }
23     return stt;
24 }
