01 // CodeStructure: Generates a broad overview over the block structure of C-family source code
02 // sourceCode: codefile to be surveyed
03 // blockCharacters: blockCharacters
04 // structure: structure
05 // index: index
06 // character: character
07  
08 public static IEnumerable<char> CodeStructure(string sourceCode)
09 {
10     List<char> blockCharacters = new List<char> { '{', '}' };
11     List<char> structure = new List<char>();
12 
13     for (int index = 0; index < sourceCode.Length; index++)
14     {
15         char character = sourceCode[index];
16         if (blockCharacters.Contains(character))
17         {
18             blockCharacters.Add(character);
19         }
20     }
21     return structure;
22 }
