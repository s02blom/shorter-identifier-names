01 // SumScores: Sums the scores of a collection of bowling games per player
02 // games: collection of scores from bowling games as comma separated integers
03 // result: result
04 // playerScores: playerScores
05 // scores: scores
06 // roll: roll
07 // lastIndex: lastIndex
08  
09 public static IEnumerable<int> SumScores(string[] games)
10 {
11     List<int> result = new List<int>();
12     foreach (string playerScores in games)
13     {
14         result.Add(0);
15         string[] scores = playerScores.Split(',');
16         foreach (string roll in scores)
17         {
18             int lastIndex = result.Count - 1;
19             games[lastIndex] += int.Parse(roll);
20         }
21     }
22     return result;
23 }
