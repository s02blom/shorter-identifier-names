01 // a: Sums the scores of a collection of bowling games per player
02 // b: collection of scores from bowling games as comma separated integers
03 // c: result
04 // d: playerScores
05 // e: scores
06 // f: roll
07 // g: lastIndex
08  
09 public static IEnumerable<int> a(string[] b)
10 {
11     List<int> c = new List<int>();
12     foreach (string d in b)
13     {
14         c.Add(0);
15         string[] e = d.Split(',');
16         foreach (string f in e)
17         {
18             int g = c.Count - 1;
19             b[g] += int.Parse(f);
20         }
21     }
22     return c;
23 }
