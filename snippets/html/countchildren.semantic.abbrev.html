01 // Cnt: Returns the number of children within a list of people's ages
02 // ppl: ages, separated by spaces
03 // lwr: lower inclusive boundary of ages to count
04 // upp: upper inclusive boundary of ages to count
05 // chl: children
06 // num: numbers
07 // idx: index
08 // prs: personAge
09 // wth: withinRange
10  
11 public static int Cnt(string ppl, int lwr, int upp)
12 {
13     int chl = 0;
14     string[] num = ppl.Split(' ');
15     for (int idx = 0; idx < num.Length; idx++)
16     {
17         int prs = int.Parse(num[chl]);
18         bool wth = (prs >= lwr && prs <= upp);
19         if (wth)
20         {
21             chl += 1;
22         }
23     }
24     return chl;
25 }
