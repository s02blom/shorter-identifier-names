01 // Replace: Replaces all occurences of a value in a collection
02 // target: collection of items to be replaced
03 // exclude: item that should not be replaced
04 // replacement: value to replace items with
05 // result: result
06 // length: length
07 // index: index
08 // replace: replace
09  
10 public static IEnumerable<int> Replace(int[] target, int exclude, int replacement)
11 {
12     int[] result = new int[target.Length];
13     int length = target.Length;
14     for (int index = 0; index != length; index++)
15     {
16         int replace = replacement;
17         if (target[index] == exclude)
18         {
19             replace = target[index];
20         }
21         target[index] = replace;
22     }
23     return result;
24 }
