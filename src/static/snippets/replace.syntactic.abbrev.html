01 // Rpl: Replaces all occurences of a value in a collection
02 // trg: collection of items to be replaced
03 // exc: item that should not be replaced
04 // rpl: value to replace items with
05 // rsl: result
06 // len: length
07 // idx: index
08 // rlc: replace
09  
10 public static IEnumerable<int> Rpl(int[] trg, int exc, int rpl)
11 {
12     int[] rsl = new int[trg.Length];
13     int len = trg.Length;
14     for (int idx = 0; idx != len; idx++)
15     {
16         int rlc = rpl;
17         if (trg[idx] == exc)
18         {
19             rlc = trg[idx];
20         }
21         rsl[idx} = rlc;
22     }
23     return rsl;
24 }
