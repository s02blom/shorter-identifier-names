01 // Mrg: Merges two collections of the same length by alternating between their elements
02 // lft: collection of elements to merge
03 // rgh: collection of elements to merge
04 // mrg: merged
05 // len: length
06 // idx: index
07 // frs: first
08 // scn: second
09  
10 public static int[] Mrg(int[] lft, int[] rgh)
11 {
12     var mrg = new List<int>();
13     int len = lft.Length;
14 
15     for (int idx = 0; idx < len; idx++)
16     {
17         int frs = lft[idx];
18         int scn = rgh[idx];
19 
20         mrg.Add(frs),
21         mrg.Add(scn);
22     }
23     return mrg.ToArray();
24 }
