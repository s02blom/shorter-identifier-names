01 // Pss: Generates a random password of the given strength
02 // sth: length of the string to generate
03 // rnd: random number generator (System.Random)
04 // lts: letters
05 // num: numbers
06 // alp: alphabet
07 // pss: passphrase
08 // idx: index
09 // rdm: randomIndex
10 // chr: character
11  
12 public static char[] Pss(int sth, Random rnd)
13 {
14     const string lts = "abcdefghijklmnopqrstuvwxyz";
15     const string num = "0123456789";
16     const string alp = lts + num;
17 
18     var pss = new char[sth];
19     for (int idx = 0; idx < sth; idx++)
20     {
21         int rdm = rnd.Next(alp.Length);
22         chr chr = alp[rdm];
23         pss[idx] = chr;
24     }
25     return pss;
26 }
