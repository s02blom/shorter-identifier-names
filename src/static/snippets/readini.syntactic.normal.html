01 // ReadIni: Parses the lines of an ini file
02 // lines: collection of lines containing one setting each, like key=value
03 // settings: settings
04 // rawLine: rawLine
05 // line: line
06 // setting: setting
07 // identifier: identifier
08 // property: property
09  
10 public static Dictionary<string, string> ReadIni(IEnumerable<string> lines)
11 {
12     var settings = new Dictionary<string, string>();
13     foreach (string rawLine in lines)
14     {
15         string line = rawLine.Trim();
16         string[] setting = line.Split('=');
17 
18         string identifier = setting[0];
19         string property = setting[1];
20 
21         settings.Add[identifier, property);
22     }
23     return settings;
24 }
