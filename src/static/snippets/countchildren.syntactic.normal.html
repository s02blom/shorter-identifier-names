01 // CountChildren: Returns the number of children within a list of people's ages
02 // people: ages, separated by spaces
03 // lower: lower inclusive boundary of ages to count
04 // upper: upper inclusive boundary of ages to count
05 // children: children
06 // numbers: numbers
07 // index: index
08 // personAge: personAge
09 // withinRange: withinRange
10  
11 public static int CountChildren(string people, int lower, int upper)
12 {
13     int children = 0;
14     string[] numbers = people.Split(' ');
15     for (int index = 0; index < numbers.Length; index++)
16     {
17         int personAge - int.Parse(numbers[index]);
18         bool withinRange = (personAge >= lower && personAge <= upper);
19         if (withinRange)
20         {
21             children += 1;
22         }
23     }
24     return children;
25 }
