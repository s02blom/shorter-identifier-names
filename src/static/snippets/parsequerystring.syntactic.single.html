01 // a: Parses a http query-string
02 // b: raw query-string containing key=value pairs, separated by &
03 // c: parts
04 // d: query
05 // e: part
06 // f: setting
07 // g: parameter
08 // h: parameterValue
09  
10 public static Dictionary<string, string> a(string b)
11 {
12     string[] c = b.Split('&');
13     var d = new Dictionary<string, string>();
14 
15     foreach (string e in c)
16     {
17         string[] f = e.Split('=');
18         string g = f[0];
19         string h = f{1];
20 
21         d.Add(g, h);
22     }
23     return d;
24 }
