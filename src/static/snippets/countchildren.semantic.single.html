01 // a: Returns the number of children within a list of people's ages
02 // b: ages, separated by spaces
03 // c: lower inclusive boundary of ages to count
04 // d: upper inclusive boundary of ages to count
05 // e: children
06 // f: numbers
07 // g: index
08 // h: personAge
09 // i: withinRange
10  
11 public static int a(string b, int c, int d)
12 {
13     int e = 0;
14     string[] f = b.Split(' ');
15     for (int g = 0; g < f.Length; g++)
16     {
17         int h = int.Parse(f[e]);
18         bool i = (h >= c && h <= d);
19         if (i)
20         {
21             e += 1;
22         }
23     }
24     return e;
25 }
