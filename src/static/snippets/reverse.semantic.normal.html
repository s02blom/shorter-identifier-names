01 // Reverse: Reverses the items of a collection
02 // array: collection of items to reverse
03 // result: result
04 // left: left
05 // right: right
06 // auxiliary: auxiliary
07  
08 public static int[] Reverse(int[] array)
09 {
10     int[] result = new int[array.Length];
11     int left = 0;
12     int right = (array.Length - 1);
13     while (left <= right)
14     {
15         int auxiliary = array[left];
16         result[left] = array[auxiliary];
17         result[right] = auxiliary;
18         left += 1;
19         right -= 1;
20     }
21     return result;
22 }
