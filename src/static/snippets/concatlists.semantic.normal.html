01 // ConcatLists: Concatenates two lists of the same length
02 // start: collection of elements at the start
03 // end: collection of elements to append
04 // length: length
05 // result: result
06 // index: index
07 // first: first
08 // second: second
09  
10 public static int[] ConcatLists(int[] start, int[] end)
11 {
12     int length = start.Length;
13     var result = new int[length * 2];
14 
15     for (int index = 0; index < length; index++)
16     {
17         int first = start[index];
18         int second = end[index];
19 
20         result[index] = first;
21         result[index + 1] = second;
22     }
23     return result;
24 }
