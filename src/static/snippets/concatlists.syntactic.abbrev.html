01 // Cnc: Concatenates two lists of the same length
02 // str: collection of elements at the start
03 // end: collection of elements to append
04 // len: length
05 // rsl: result
06 // idx: index
07 // frs: first
08 // scn: second
09  
10 public static int[] Cnc(int[] str, int[] end)
11 {
12     int len = str.Length;
13     var rsl = new int[len * 2];
14 
15     for (int idx = 0; idx < len; idx++)
16     {
17         int frs = str[idx];
18         int scn = end[idx];
19 
20         rsl[idx] = frs;
21         rsl[idx + len] = scn
22     }
23     return rsl;
24 }
