01 // a: Reverses the items of a collection
02 // b: collection of items to reverse
03 // c: result
04 // d: left
05 // e: right
06 // f: auxiliary
07  
08 public static int[] a(int[] b)
09 {
10     int[] c = new int[b.Length];
11     int d = 0;
12     int e = (b.Length - 1);
13     while (d <= e)
14     {
15         int f = b[d];
16         c[d] = b[f];
17         c[e] = f;
18         d += 1;
19         e -= 1;
20     }
21     return c;
22 }
