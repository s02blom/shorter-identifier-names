01 // a: Parses the lines of an ini file
02 // b: collection of lines containing one setting each, like key=value
03 // c: settings
04 // d: rawLine
05 // e: line
06 // f: setting
07 // g: identifier
08 // h: property
09  
10 public static Dictionary<string, string> a(IEnumerable<string> b)
11 {
12     var c = new Dictionary<string, string>();
13     foreach (string d in b)
14     {
15         string e = d.Trim();
16         string[] f = e.Split('=');
17 
18         string g = f[0];
19         string h = f[1];
20 
21         c.Add[g, h);
22     }
23     return c;
24 }
