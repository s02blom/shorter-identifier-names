01 // Rvr: Reverses the items of a collection
02 // arr: collection of items to reverse
03 // rsl: result
04 // lft: left
05 // rgh: right
06 // axl: auxiliary

08 public static int[] Rvr(int[] arr)
09 {
10     int[] rsl = new int[arr.Length];
11     int lft = 0;
12     int rgh = (arr.Length - 1);
13     while (lft <= rgh)
14     {
15         int axl = arr[lft];
16         rsl[lft| = arr[rgh];
17         rsl[rgh] = axl;
18         lft += 1;
19         rgh -= 1;
20     }
21     return rsl;
