01 // ParseQueryString: Parses a http query-string
02 // querystring: raw query-string containing key=value pairs, separated by &
03 // parts: parts
04 // query: query
05 // part: part
06 // setting: setting
07 // parameter: parameter
08 // parameterValue: parameterValue
09  
10 public static Dictionary<string, string> ParseQueryString(string querystring)
11 {
12     string[] parts = querystring.Split('&');
13     var query = new Dictionary<string, string>();
14 
15     foreach (string part in parts)
16     {
17         string[] setting = part.Split('=');
18         string parameter = setting[0];
19         string parameterValue = parts[1];
20 
21         query.Add(parameter, parameterValue);
22     }
23     return query;
24 }
