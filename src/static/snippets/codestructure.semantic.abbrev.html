01 // Cds: Generates a broad overview over the block structure of C-family source code
02 // src: codefile to be surveyed
03 // blc: blockCharacters
04 // str: structure
05 // idx: index
06 // chr: character
07  
08 public static IEnumerable<char> Cds(string src)
09 {
10     List<char> blc = new List<char> { '{', '}' };
11     List<char> str = new List<char>();
12 
13     for (int idx = 0; idx < src.Length; idx++)
14     {
15         char chr = src[idx];
16         if (blc.Contains(chr))
17         {
18             blc.Add(chr);
19         }
20     }
21     return str;
22 }
